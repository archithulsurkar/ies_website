Sample Programs

Try these sample programs to get hands-on experience with Arduino.

Blink an LED


void setup() {
  pinMode(LED_BUILTIN, OUTPUT);
}

void loop() {
  digitalWrite(LED_BUILTIN, HIGH);
  delay(1000);
  digitalWrite(LED_BUILTIN, LOW);
  delay(1000);
}

Button Controlled LED


const int buttonPin = 2;
const int ledPin = 13;
void setup() {
  pinMode(buttonPin, INPUT);
  pinMode(ledPin, OUTPUT);
}
void loop() {
  if (digitalRead(buttonPin) == HIGH) {
    digitalWrite(ledPin, HIGH);
  } else {
    digitalWrite(ledPin, LOW);
  }
}

Buzzer Tone Generation


const int buzzer = 9;
void setup() {
  pinMode(buzzer, OUTPUT);
}
void loop() {
  tone(buzzer, 1000); // 1kHz frequency
  delay(500);
  noTone(buzzer);
  delay(500);
}

Analog Sensor Reading (Potentiometer)


const int potPin = A0;
void setup() {
  Serial.begin(9600);
}
void loop() {
  int value = analogRead(potPin);
  Serial.println(value);
  delay(500);
}

Servo Motor Control


#include 
Servo myServo;
void setup() {
  myServo.attach(9);
}
void loop() {
  myServo.write(0);
  delay(1000);
  myServo.write(90);
  delay(1000);
  myServo.write(180);
  delay(1000);
}

Temperature Sensor Reading (LM35)


const int tempPin = A0;
void setup() {
  Serial.begin(9600);
}
void loop() {
  int value = analogRead(tempPin);
  float voltage = value * (5.0 / 1023.0);
  float temperature = voltage * 100;
  Serial.print("Temperature: ");
  Serial.print(temperature);
  Serial.println("°C");
  delay(1000);
}
